javascript
/ 1. DOM Manipulation: Use querySelector to get elements
const taskInput = document.querySelector('#taskInput');
const addButton = document.querySelector('#addButton');
const taskList = document.querySelector('#taskList');

// 2. Event Handling: Define the function to add a new task
function addTask() {
    const taskText = taskInput.value.trim(); // Get the text and remove whitespace

    // Check if the input is not empty
    if (taskText === '') {
        alert('Please enter a task.');
        return; // Stop the function if the input is empty
    }

    // DOM Manipulation: Create a new list item (li) and set its content
    const listItem = document.createElement('li');
    listItem.textContent = taskText;

    // DOM Manipulation: Append the new li to the ul
    taskList.appendChild(listItem);

    // Clear the input field after adding the task
    taskInput.value = '';
}

// 3. Event Handling: Attach the addTask function to the 'click' event of the button
addButton.addEventListener('click', addTask);

// Optional: Allow adding tasks by pressing the 'Enter' key in the input field
taskInput.addEventListener('keypress', function(event) {
    if (event.key === 'Enter') {
        addTask();
 addTask();
    }
});
